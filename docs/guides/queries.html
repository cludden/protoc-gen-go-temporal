Queries

Implementation​

A query is implemented as a method on Workflow struct.

  • Go
  • Schema
main.go
package main

import (
    examplev1 "path/to/gen/example/v1"
    "go.temporal.io/sdk/workflow"
)

type FooWorkflow struct {
    *examplev1.FooWorkflowInput
}

func (w *FooWorkflow) Execute(ctx workflow.Context) (*examplev1.FooOutput, error) {
    return &examplev1.FooOutput{}, nil
}

func (w *FooWorkflow) Bar(input *examplev1.BarInput) (*examplev1.BarOutput, error) {
    return &examplev1.BarOutput{}, nil
}
example.proto
syntax="proto3";

package example.v1;

import "temporal/v1/temporal.proto";

service Example {
  rpc Foo(FooInput) returns (FooOutput) {
    option (temporal.v1.workflow) = {
      query: { ref: 'Bar' }
    };
  }

  rpc Bar(BarInput) returns (BarOutput) {
    option (temporal.v1.query) = {};
  }
}

Parameters​

The signature of a Query method varies based on whether or not the query specifies a non-empty input message type.

warning

Queries are required to specify a non-empty output type.

A query input can be omitted using the native google.protobuf.Empty type. This modifies the signature of the Query method to accept no arguments.

note

This requires an additional google/protobuf/empty.proto protobuf import statement.

example.proto
syntax="proto3";

package example.v1;

import "temporal/v1/temporal.proto";

service Example {
  rpc Foo(FooInput) returns (FooOutput) {
    option (temporal.v1.workflow) = {
      query: { ref: 'Bar' }
    };
  }

  rpc Bar(google.protobuf.Empty) returns (BarOutput) {
    option (temporal.v1.query) = {};
  }
}
main.go

    examplev1 "path/to/gen/example/v1"

func (w *FooWorkflow) Execute(ctx workflow.Context) (*examplev1.FooOutput, error) {
    return &examplev1.FooOutput{}, nil
}

func (w *FooWorkflow) Bar() (*examplev1.BarOutput, error) {
    return &examplev1.BarOutput{}, nil
}

Invocation​

This plugin supports several methods for executing protobuf queries, each of which is outlined in more detail below.

Client​

Consumers can utilize the generated Client to execute queries from any Go application. See the Clients guide for more usage details.

  • Go
  • Schema
main.go
package main

import (
    "context"
    "log"

    examplev1 "path/to/gen/example/v1"
    "go.temporal.io/sdk/client"
)

func main() {
    // initialize temporal client
    c, err := client.Dial(client.Options{})
    if err != nil {
        log.Fatalf("error initializing client: %v", err)
    }

    // initialize temporal protobuf client
    client := examplev1.NewExampleClient(c)

    // start an example.v1.Example.Foo workflow
    run, err := client.FooAsync(context.Background(), &examplev1.FooInput{})
    if err != nil {
        log.Fatalf("error starting example.v1.Example.Foo workflow: %v", err)
    }

    // query the workflow
    bar, err := run.Bar(context.Background(), &examplev1.BarInput{})
    if err != nil {
        log.Fatalf("error executing example.v1.Example.Bar query for workflow %q: %v", run.ID(), err)
    }

    // wait for workflow completion
    foo, err := run.Get(context.Background())
    if err != nil {
        log.Fatalf("error executing example.v1.Example.Foo workflow: %v", err)
    }
}
example.proto
syntax="proto3";

package example.v1;

import "temporal/v1/temporal.proto";

service Example {

  rpc Foo(FooInput) returns (FooOutput) {

      query: { ref: 'Bar' }
    };
  }

  // Bar retrieves the status of a foo operation
  rpc Bar(BarInput) returns (BarOutput) {
    option (temporal.v1.query) = {};
  }
}

main.go
package main

import (
    "log"
    "os"

    examplev1 "path/to/gen/example/v1"
)

func main() {
    app, err := examplev1.NewExampleCLI()
    if err != nil {
        log.Fatalf("error initializing cli: %v", err)
    }

    if err := app.Run(os.Args); err != nil {
        log.Fatal(err)
    }
}
example.proto
syntax="proto3";

package example.v1;

import "temporal/v1/temporal.proto";

service Example {
  // Foo performs a foo operation
  rpc Foo(FooInput) returns (FooOutput) {
    option (temporal.v1.workflow) = {
      id: 'foo/${! name }'
      query: { ref: 'Bar' }
    };
  }

  // Bar retrieves the status of a foo operation
  rpc Bar(BarInput) returns (BarOutput) {
    option (temporal.v1.query) = {};
  }
}

message FooInput {
  string name = 1;
}

message FooOutput {}

message BarInput {
  bool verbose = 1;
}

message BarOutput {
  bool ok = 1;
}

Cross-Namespace (XNS)​

Queries can be executed from other workflows in a different Temporal namespace or even an entirely separate Temporal cluster (e.g. on-prem to cloud). See the Cross-Namespace guide for more usage details.

  • Go
  • Schema
example.go
package example

import (
    "fmt"

    examplev1 "path/to/gen/example/v1"
    "path/to/gen/example/v1/examplev1xns"
    "go.temporal.io/sdk/workflow"
)

func MyWorkflow(ctx workflow.Context) error {
    run, err := examplev1xns.FooAsync(ctx, &examplev1.FooInput{})
    if err != nil {
        return fmt.Errorf("error staring example.v1.Example.Foo xns workflow: %w", err)
    }

    bar, err := run.Bar(ctx, &examplev1.BarInput{})
    if err != nil {
        return fmt.Errorf("error querying xns workflow: %w", err)
    }

    foo, err := run.Get(ctx)
    if err != nil {
        return fmt.Errorf("error executing example.v1.Example.Foo xns workflow")
    }
    return nil
}
